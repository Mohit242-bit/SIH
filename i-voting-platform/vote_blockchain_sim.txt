import hashlib
import json
import time

class Block:
    """A single block in the blockchain."""
    def __init__(self, index, voter_id, candidate, previous_hash):
        self.index = index
        self.timestamp = time.time()
        self.voter_id = voter_id
        self.candidate = candidate
        self.previous_hash = previous_hash
        # The hash is calculated after all other data is set
        self.hash = self.calculate_hash()

    def calculate_hash(self):
        """Calculates the cryptographic hash of the block."""
        # Create a dictionary of the block's data, excluding the hash itself.
        block_data = {
            "index": self.index,
            "timestamp": self.timestamp,
            "voter_id": self.voter_id,
            "candidate": self.candidate,
            "previous_hash": self.previous_hash
        }
        # Dump the dictionary to a string and then encode it to bytes
        block_string = json.dumps(block_data, sort_keys=True)
        return hashlib.sha256(block_string.encode()).hexdigest()

class E_Voting_Blockchain:
    """The blockchain ledger for our e-voting system."""
    def __init__(self):
        self.chain = [self.create_genesis_block()]
        self.voted_voters = set()  # To store IDs of voters who have already voted.
        # This list simulates a database of valid, registered voters
        self.valid_voters = ["Voter_001", "Voter_002", "Voter_003"]

    def create_genesis_block(self):
        """Creates the first block in the chain."""
        return Block(0, "System", "Election Started", "0")
    def get_vote_tally(self):
        """Counts the votes and displays the current results."""
        vote_count = {}
        for block in self.chain:
            if block.index > 0:
                candidate = block.candidate
                vote_count[candidate] = vote_count.get(candidate, 0) + 1
        return vote_count

    def get_last_block(self):
        """Returns the most recent block in the chain."""
        return self.chain[-1]

    def cast_vote(self, voter_id, candidate):
        """Casts a vote after checking if the voter is valid and hasn't voted yet."""
        # Check 1: Is the voter ID in our "database" of valid voters?
        if voter_id not in self.valid_voters:
            print(f"Error: Voter with ID '{voter_id}' is not a registered voter. 🚫")
            return

        # Check 2: Has the voter already cast a vote in this election?
        if voter_id in self.voted_voters:
            print(f"Error: Voter with ID '{voter_id}' has already voted. ❌")
            return

        print(f"Casting vote for candidate '{candidate}'...")
        new_block = Block(
            index=len(self.chain),
            voter_id=voter_id,
            candidate=candidate,
            previous_hash=self.get_last_block().hash
        )
        self.chain.append(new_block)
        self.voted_voters.add(voter_id)
        print("Vote successfully added to the blockchain. ✅")

    def is_chain_valid(self):
        """Checks if the entire blockchain is valid and untampered."""
        for i in range(1, len(self.chain)):
            current_block = self.chain[i]
            previous_block = self.chain[i-1]
            
            # Check 1: Verify the block's own hash against its content
            if current_block.hash != current_block.calculate_hash():
                print(f"Chain corrupted: Block {current_block.index} has an invalid hash.")
                return False

            # Check 2: Verify that the blocks are correctly linked
            if current_block.previous_hash != previous_block.hash:
                print(f"Chain corrupted: Block {current_block.index} is not linked correctly.")
                return False
        return True
